## 🔍 技术方案对比

### 方案 A：Fuse.js (模糊搜索)

**GitHub**: https://github.com/krisk/Fuse

**特点**:
- ⭐ 65k+ stars
- 📦 纯 JavaScript，无依赖
- 🎯 模糊搜索专家
- 🌏 支持多语言

**示例**:
```javascript
import Fuse from 'fuse.js';

const fuse = new Fuse(documents, {
  keys: ['title', 'body'],
  threshold: 0.3,  // 0.0 = 精确，1.0 = 任意匹配
  distance: 100,   // 匹配距离
  minMatchCharLength: 2,
  includeMatches: true,
});

const results = fuse.search('内存安权');
// 返回：[{ item: {...}, matches: [...] }]
```

**开销**:
- 库大小：~30KB (gzipped)
- 内存：20-50MB (5k 行文本)
- 搜索速度：50-200ms

### 方案 B：Minisearch (全文搜索 + 模糊)

**GitHub**: https://github.com/lucaong/minisearch

**特点**:
- ⭐ 5.8k+ stars
- 📦 超轻量 (~7KB)
- 🎯 支持模糊搜索
- 🔧 可配置前缀搜索

**示例**:

```javascript
import MiniSearch from 'minisearch';

const miniSearch = new MiniSearch({

```

**开销**:
- 库大小：~7KB (gzipped)
- 内存：10-30MB (5k 行文本)
- 搜索速度：30-100ms

### 方案 C：自定义正则搜索

**实现方式**:
```javascript
function regexSearch(pattern, documents) {
  try {
    const regex = new RegExp(pattern, 'gi');
    return documents.filter(doc => 
      regex.test(doc.title) || regex.test(doc.body)
    );
  } catch (e) {
    // 无效正则表达式
    return [];
  }
}
```

**开销**:
- 库大小：0KB (原生支持)
- 内存：5-10MB
- 搜索速度：20-100ms (5k 行)

## 💻 实现方案

### 阶段 1：添加模糊搜索

#### 1.1 集成 Fuse.js

**修改 searcher.js**:
```javascript
// 在 searcher.js 开头添加
let fuseIndex = null;
let fuse = null;

// 加载索引时同时构建 Fuse 索引
function loadSearchIndex() {
    // 原有 elasticlunr 索引
    searchindex = elasticlunr.Index.load(window.search.index);
    
    // 新建 Fuse 索引
    const documents = Object.values(searchindex.documentStore.docs).map((doc, id) => ({
        id: id,
        title: doc.title,
        body: doc.body,
        breadcrumbs: doc.breadcrumbs,
    }));
    
    fuse = new Fuse(documents, {
        keys: [
            { name: 'title', weight: 2 },
            { name: 'body', weight: 1 },
            { name: 'breadcrumbs', weight: 0.5 },
        ],
        threshold: 0.3,  // 允许 30% 差异
        distance: 100,   // 匹配距离
        minMatchCharLength: 2,
        includeMatches: true,
        useExtendedSearch: true,
    });
    
    fuseIndex = documents;
}
```

#### 1.2 模糊搜索函数

```javascript
function fuzzySearch(query) {
    // 1. 先用 elasticlunr 搜索（精确匹配）
    const exactResults = searchindex.search(query, search_options);
    
    // 2. 如果结果少，启用模糊搜索
    if (exactResults.length < 3) {
        const fuzzyResults = fuse.search(query).map(result => ({
            ref: result.item.id,
            score: result.score,
            doc: searchindex.documentStore.docs[result.item.id],
        }));
        
        // 3. 合并结果
        return mergeResults(exactResults, fuzzyResults);
    }
    
    return exactResults;
}

function mergeResults(exact, fuzzy) {
    const merged = new Map();
    
    // 添加精确结果
    exact.forEach(r => merged.set(r.ref, r));
    
    // 添加模糊结果（去重）
    fuzzy.forEach(r => {
        if (!merged.has(r.ref)) {
            merged.set(r.ref, r);
        }
    });
    
    // 按分数排序
    return Array.from(merged.values())
        .sort((a, b) => b.score - a.score);
}
```

### 阶段 2：添加正则搜索

#### 2.1 正则搜索函数

```javascript
function regexSearch(pattern) {
    try {
        // 移除前导 "/" 如果存在
        if (pattern.startsWith('/')) {
            pattern = pattern.slice(1);
        }
        
        // 支持尾部标志
        let flags = 'gi';
        if (/[imsux]+$/.test(pattern)) {
            const match = pattern.match(/([imsux]+)$/);
            flags = match[1];
            pattern = pattern.slice(0, -match[1].length);
        }
        
        const regex = new RegExp(pattern, flags);
        
        // 在所有文档中搜索
        const docs = searchindex.documentStore.docs;
        const results = [];
        
        for (const [ref, doc] of Object.entries(docs)) {
            if (regex.test(doc.title) || 
                regex.test(doc.body) || 
                regex.test(doc.breadcrumbs)) {
                
                // 找到所有匹配
                const matches = [];
                const allText = `${doc.title} ${doc.body}`;
                let match;
                regex.lastIndex = 0;
                while ((match = regex.exec(allText)) !== null) {
                    matches.push(match[0]);
                }
                
                results.push({
                    ref: ref,
                    score: matches.length,
                    doc: doc,
                    matches: matches.slice(0, 5),  // 限制显示数量
                });
            }
        }
        
        return results.sort((a, b) => b.score - a.score);
        
    } catch (e) {
        console.error('Invalid regex:', e);
        return [];
    }
}
```

#### 2.2 搜索模式检测

```javascript
function detectSearchMode(query) {
    // 正则模式：以 "/" 开头
    if (query.startsWith('/')) {
        return 'regex';
    }
    
    // 模糊模式：包含明显拼写错误特征
    // - 连续重复字符："mmemroy"
    // - 常见错别字模式
    // - 拼音输入特征
    if (hasTypos(query)) {
        return 'fuzzy';
    }
    
    // 默认：精确模式
    return 'exact';
}

function hasTypos(query) {
    // 检测连续重复字符
    if (/(.)\1{2,}/.test(query)) return true;
    
    // 检测常见错别字
    const commonTypos = {
        '安权': '安全',
        '编成': '编程',
        '总节': '总结',
        // ... 可以扩展
    };
    
    for (const typo in commonTypos) {
        if (query.includes(typo)) return true;
    }
    
    return false;
}
```

### 阶段 3：统一搜索接口

```javascript
// 主搜索函数
function search(query) {
    const mode = detectSearchMode(query);
    
    let results;
    switch (mode) {
        case 'regex':
            results = regexSearch(query);
            break;
        case 'fuzzy':
            results = fuzzySearch(query);
            break;
        default:
            results = searchindex.search(query, search_options);
    }
    
    // 显示结果
    displayResults(results, mode);
    
    return results;
}
```
