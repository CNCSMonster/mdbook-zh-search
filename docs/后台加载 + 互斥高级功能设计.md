# 后台加载 + 互斥高级功能设计

## 📋 核心问题

1. **2 秒加载时间可以接受** → 改为后台静默加载
2. **正则搜索和模糊搜索冲突** → 只能启用一个

---

## 🎯 优化方案

### 方案 A：后台静默加载

```
页面加载时:
├── 立即：基础搜索 (searchindex.js) - 2 秒
├── 后台：jieba-wasm - 1 秒 (用户无感知)
├── 后台：fuzzy_search.js - 0.5 秒 (用户无感知)
└── 后台：regex_bg.wasm - 0.5 秒 (用户无感知)

用户点击"模糊搜索":
└── 立即启用 (已加载完成) ✅

用户点击"正则搜索":
└── 切换模式，立即启用 ✅
```

**优势**:
- ✅ 用户体验好（无需等待）
- ✅ 总加载时间不变（后台并行）
- ✅ 功能切换无延迟

---

### 方案 B：互斥的高级功能

```
搜索模式:
├── 精确模式 (默认)
├── 模糊模式 (与正则互斥)
└── 正则模式 (与模糊互斥)

UI 设计:
[🔍 精确] [✨ 模糊] [.* 正则]

点击"模糊":
├── 如果"正则"已启用 → 先关闭正则
└── 启用模糊

点击"正则":
├── 如果"模糊"已启用 → 先关闭模糊
└── 启用正则
```

## 🏗️ 架构设计

### 状态机

```javascript
const SearchMode = {
    EXACT: 'exact',     // 精确搜索
    FUZZY: 'fuzzy',     // 模糊搜索
    REGEX: 'regex',     // 正则搜索
};

class SearchManager {
    constructor() {
        this.mode = SearchMode.EXACT;
        this.modules = {
            jieba: null,      // 后台加载
            fuzzy: null,      // 后台加载
            regex: null,      // 后台加载
        };
        this.loadingState = {
            jieba: 'loading',  // 'loading' | 'ready' | 'error'
            fuzzy: 'loading',
            regex: 'loading',
        };
    }
    
    // 后台静默加载所有模块
    async backgroundLoad() {
        // 并行加载，不阻塞 UI
        Promise.all([
            this.loadJiebaWasm(),
            this.loadFuzzyModule(),
            this.loadRegexModule(),
        ]).then(() => {
            console.log('所有搜索模块加载完成');
            this.updateUI();
        }).catch(error => {
            console.error('模块加载失败:', error);
        });
    }
    
    async loadJiebaWasm() {
        try {
            const module = await import('./jieba_wasm.js');
            this.modules.jieba = module;
            this.loadingState.jieba = 'ready';
        } catch (e) {
            this.loadingState.jieba = 'error';
        }
    }
    
    async loadFuzzyModule() {
        try {
            const module = await import('./fuzzy_search.js');
            this.modules.fuzzy = module;
            this.loadingState.fuzzy = 'ready';
        } catch (e) {
            this.loadingState.fuzzy = 'error';
        }
    }
    
    async loadRegexModule() {
        try {
            const wasm = await this.loadWasm('./wasm/regex_bg.wasm');
            this.modules.regex = wasm;
            this.loadingState.regex = 'ready';
        } catch (e) {
            this.loadingState.regex = 'error';
        }
    }
    
    // 切换搜索模式（互斥）
    async setMode(newMode) {
        // 如果已经是当前模式，切换到精确
        if (this.mode === newMode) {
            this.mode = SearchMode.EXACT;
            this.updateButtons();
            return;
        }
        
        // 互斥检查：模糊和正则不能同时启用
        if (newMode === SearchMode.FUZZY && this.loadingState.fuzzy !== 'ready') {
            this.showToast('模糊搜索加载中，请稍候...');
            return;
        }
        
        if (newMode === SearchMode.REGEX && this.loadingState.regex !== 'ready') {
            this.showToast('正则搜索加载中，请稍候...');
            return;
        }
        
        // 切换模式
        this.mode = newMode;
        this.updateButtons();
        this.research();
    }
    
    updateButtons() {
        const buttons = {
            [SearchMode.EXACT]: document.getElementById('btn-exact'),
            [SearchMode.FUZZY]: document.getElementById('btn-fuzzy'),
            [SearchMode.REGEX]: document.getElementById('btn-regex'),
        };
        
        // 更新按钮状态
        Object.entries(buttons).forEach(([mode, btn]) => {
            if (mode === this.mode) {
                btn.classList.add('active');
            } else {
                btn.classList.remove('active');
            }
            
            // 显示加载状态
            if (mode === SearchMode.FUZZY && this.loadingState.fuzzy === 'loading') {
                btn.textContent = '✨ 加载中...';
                btn.disabled = true;
            } else if (mode === SearchMode.REGEX && this.loadingState.regex === 'loading') {
                btn.textContent = '.* 加载中...';
                btn.disabled = true;
            } else {
                buttons[mode].disabled = false;
                if (mode === SearchMode.FUZZY) {
                    buttons[mode].textContent = '✨ 模糊';
                } else if (mode === SearchMode.REGEX) {
                    buttons[mode].textContent = '.* 正则';
                }
            }
        });
    }
    
    // 搜索
    async search(query) {
        if (!query) {
            this.hideResults();
            return;
        }
        
        let results;
        
        switch (this.mode) {
            case SearchMode.FUZZY:
                if (!this.modules.fuzzy) {
                    console.warn('模糊搜索模块未加载');
                    return;
                }
                results = await this.modules.fuzzy.fuzzySearch(query);
                break;
                
            case SearchMode.REGEX:
                if (!this.modules.regex) {
                    console.warn('正则搜索模块未加载');
                    return;
                }
                results = this.modules.regex.search(query);
                break;
                
            default:
                // 精确搜索
                results = this.index.search(query, options);
        }
        
        this.displayResults(results);
    }
}

// 启动
const searchManager = new SearchManager();
searchManager.init();  // 初始化基础搜索
searchManager.backgroundLoad();  // 后台加载所有模块
```

### searcher.js

```javascript
'use strict';

/* global elasticlunr, Mark */

// 搜索模式枚举
const SearchMode = {
    EXACT: 'exact',
    FUZZY: 'fuzzy',
    REGEX: 'regex',
};

// 搜索管理器
class SearchManager {
    constructor() {
        this.mode = SearchMode.EXACT;
        this.index = null;
        this.modules = {
            jieba: null,
            fuzzy: null,
            regex: null,
        };
        this.loadingState = {
            jieba: 'loading',
            fuzzy: 'loading',
            regex: 'loading',
        };
        
        // UI 元素
        this.searchInput = null;
        this.searchResults = null;
        this.buttons = {};
    }
    
    // 初始化
    async init() {
        // 加载基础索引
        this.index = elasticlunr.Index.load(window.search.index);
        
        // 获取 UI 元素
        this.searchInput = document.getElementById('search-input');
        this.searchResults = document.getElementById('search-results');
        this.buttons = {
            [SearchMode.EXACT]: document.getElementById('btn-exact'),
            [SearchMode.FUZZY]: document.getElementById('btn-fuzzy'),
            [SearchMode.REGEX]: document.getElementById('btn-regex'),
        };
        
        // 绑定事件
        this.bindEvents();
        
        // 后台加载所有模块
        this.backgroundLoad();
        
        console.log('[Search] 基础搜索已初始化，高级功能后台加载中...');
    }
    
    // 绑定事件
    bindEvents() {
        // 搜索输入
        this.searchInput.addEventListener('input', (e) => {
            this.search(e.target.value);
        });
        
        // 模式切换
        Object.entries(this.buttons).forEach(([mode, btn]) => {
            btn.addEventListener('click', () => {
                this.setMode(mode);
            });
        });
        
        // 键盘快捷键
        document.addEventListener('keydown', (e) => {
            if (e.key === '/' && !this.isInputFocused()) {
                e.preventDefault();
                this.searchInput.focus();
            }
        });
    }
    
    // 后台静默加载所有模块
    async backgroundLoad() {
        const loadingPromises = [
            this.loadJiebaWasm(),
            this.loadFuzzyModule(),
            this.loadRegexModule(),
        ];
        
        Promise.allSettled(loadingPromises).then((results) => {
            const success = results.filter(r => r.status === 'fulfilled').length;
            const total = results.length;
            console.log(`[Search] 后台加载完成：${success}/${total} 模块成功`);
            
            // 更新 UI
            this.updateButtons();
            
            // 隐藏加载指示器
            const indicator = document.getElementById('loading-indicator');
            if (indicator) {
                indicator.classList.add('hidden');
            }
        });
    }
    
    async loadJiebaWasm() {
        try {
            const module = await import('./jieba_wasm.js');
            this.modules.jieba = module;
            this.loadingState.jieba = 'ready';
            console.log('[Search] jieba-wasm 加载完成');
        } catch (e) {
            this.loadingState.jieba = 'error';
            console.error('[Search] jieba-wasm 加载失败:', e);
        }
    }
    
    async loadFuzzyModule() {
        try {
            const module = await import('./fuzzy_search.js');
            this.modules.fuzzy = module;
            this.loadingState.fuzzy = 'ready';
            console.log('[Search] 模糊搜索模块加载完成');
        } catch (e) {
            this.loadingState.fuzzy = 'error';
            console.error('[Search] 模糊搜索模块加载失败:', e);
        }
    }
    
    async loadRegexModule() {
        try {
            const wasm = await this.loadWasm('./wasm/regex_bg.wasm');
            this.modules.regex = wasm;
            this.loadingState.regex = 'ready';
            console.log('[Search] 正则搜索模块加载完成');
        } catch (e) {
            this.loadingState.regex = 'error';
            console.error('[Search] 正则搜索模块加载失败:', e);
        }
    }
    
    async loadWasm(path) {
        const response = await fetch(path);
        const { instance } = await WebAssembly.instantiateStreaming(response);
        return instance.exports;
    }
    
    // 切换搜索模式（互斥）
    setMode(newMode) {
        // 如果已经是当前模式，切换到精确
        if (this.mode === newMode) {
            this.mode = SearchMode.EXACT;
            this.updateButtons();
            this.search(this.searchInput.value);
            return;
        }
        
        // 检查模块是否已加载
        if (newMode === SearchMode.FUZZY) {
            if (this.loadingState.fuzzy === 'loading') {
                this.showToast('✨ 模糊搜索加载中，请稍候...');
                return;
            }
            if (this.loadingState.fuzzy === 'error') {
                this.showToast('❌ 模糊搜索加载失败');
                return;
            }
        }
        
        if (newMode === SearchMode.REGEX) {
            if (this.loadingState.regex === 'loading') {
                this.showToast('.* 正则搜索加载中，请稍候...');
                return;
            }
            if (this.loadingState.regex === 'error') {
                this.showToast('❌ 正则搜索加载失败');
                return;
            }
        }
        
        // 切换模式（自动关闭另一个高级功能）
        this.mode = newMode;
        this.updateButtons();
        this.search(this.searchInput.value);
        
        console.log(`[Search] 切换到 ${newMode} 模式`);
    }
    
    // 搜索
    async search(query) {
        if (!query) {
            this.hideResults();
            return;
        }
        
        let results;
        
        try {
            switch (this.mode) {
                case SearchMode.FUZZY:
                    if (!this.modules.fuzzy) {
                        console.warn('[Search] 模糊搜索模块未就绪');
                        return;
                    }
                    results = await this.modules.fuzzy.fuzzySearch(query);
                    break;
                    
                case SearchMode.REGEX:
                    if (!this.modules.regex) {
                        console.warn('[Search] 正则搜索模块未就绪');
                        return;
                    }
                    results = this.modules.regex.search(query);
                    break;
                    
                default:
                    // 精确搜索
                    results = this.index.search(query, {
                        bool: 'AND',
                        expand: true,
                        fields: {
                            title: {boost: 2},
                            body: {boost: 1},
                        },
                    });
            }
            
            this.displayResults(results);
            
        } catch (e) {
            console.error('[Search] 搜索失败:', e);
            this.showError('搜索失败，请重试');
        }
    }
    
    // 更新按钮状态
    updateButtons() {
        Object.entries(this.buttons).forEach(([mode, btn]) => {
            // 更新激活状态
            if (mode === this.mode) {
                btn.classList.add('active');
            } else {
                btn.classList.remove('active');
            }
            
            // 更新加载状态
            if (this.loadingState[mode] === 'loading') {
                btn.disabled = true;
                btn.classList.add('loading');
                btn.textContent = this.getLoadingText(mode);
            } else if (this.loadingState[mode] === 'error') {
                btn.disabled = true;
                btn.classList.remove('loading');
                btn.textContent = this.getErrorText(mode);
            } else {
                btn.disabled = false;
                btn.classList.remove('loading');
                btn.textContent = this.getDefaultText(mode);
            }
        });
    }
    
    getLoadingText(mode) {
        switch (mode) {
            case SearchMode.FUZZY: return '✨ 加载中...';
            case SearchMode.REGEX: return '.* 加载中...';
            default: return '🔍 精确';
        }
    }
    
    getErrorText(mode) {
        switch (mode) {
            case SearchMode.FUZZY: return '❌ 失败';
            case SearchMode.REGEX: return '❌ 失败';
            default: return '🔍 精确';
        }
    }
    
    getDefaultText(mode) {
        switch (mode) {
            case SearchMode.FUZZY: return '✨ 模糊';
            case SearchMode.REGEX: return '.* 正则';
            default: return '🔍 精确';
        }
    }
    
    // 重新搜索（切换模式后）
    research() {
        this.search(this.searchInput.value);
    }
    
    // 显示结果
    displayResults(results) {
        if (!results || results.length === 0) {
            this.showNoResults();
            return;
        }
        
        // 显示结果数量
        const countElem = document.getElementById('search-count');
        if (countElem) {
            countElem.textContent = `找到 ${results.length} 个结果`;
        }
        
        // 渲染结果列表
        this.searchResults.innerHTML = results.map(result => `
            <li class="search-result-item">
                <a href="${result.doc.url}">
                    <div class="result-title">${this.highlight(result.doc.title)}</div>
                    <div class="result-breadcrumbs">${result.doc.breadcrumbs}</div>
                </a>
            </li>
        `).join('');
        
        this.searchResults.parentElement.classList.remove('hidden');
    }
    
    showNoResults() {
        const countElem = document.getElementById('search-count');
        if (countElem) {
            countElem.textContent = '未找到结果';
        }
        this.searchResults.innerHTML = '<li class="no-results">没有找到匹配的结果</li>';
        this.searchResults.parentElement.classList.remove('hidden');
    }
    
    hideResults() {
        this.searchResults.parentElement.classList.add('hidden');
    }
    
    showError(message) {
        this.searchResults.innerHTML = `<li class="error">${message}</li>`;
        this.searchResults.parentElement.classList.remove('hidden');
    }
    
    showToast(message) {
        // 简单的 toast 提示
        const toast = document.createElement('div');
        toast.className = 'search-toast';
        toast.textContent = message;
        toast.style.cssText = `
            position: fixed;
            bottom: 20px;
            left: 50%;
            transform: translateX(-50%);
            background: rgba(0,0,0,0.8);
            color: white;
            padding: 8px 16px;
            border-radius: 4px;
            z-index: 1000;
        `;
        document.body.appendChild(toast);
        setTimeout(() => toast.remove(), 3000);
    }
    
    isInputFocused() {
        return document.activeElement === this.searchInput;
    }
    
    highlight(text) {
        // 高亮关键词
        return text;
    }
}

// 启动搜索管理器
window.searchManager = new SearchManager();
if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', () => {
        window.searchManager.init();
    });
} else {
    window.searchManager.init();
}
```

## 💡 关键设计决策

### 决策 1：后台静默加载

**理由**:
- 2 秒加载时间可接受
- 后台并行加载，用户无感知
- 功能切换无需等待

**实现**:
```javascript
// 不阻塞 UI
this.backgroundLoad();

// 并行加载所有模块
Promise.allSettled([
    this.loadJiebaWasm(),
    this.loadFuzzyModule(),
    this.loadRegexModule(),
]);
```

---

### 决策 2：互斥的高级功能

**理由**:
- 模糊搜索和正则搜索算法冲突
- 用户通常只需要一种高级功能
- 减少内存占用

**实现**:
```javascript
setMode(newMode) {
    // 互斥：只能启用一个高级功能
    this.mode = newMode;
    this.updateButtons();
    this.search(this.searchInput.value);
}
```

---

### 决策 3：优雅的降级

**如果模块加载失败**:
```javascript
if (this.loadingState.fuzzy === 'error') {
    this.showToast('❌ 模糊搜索加载失败');
    // 自动切换回精确搜索
    this.mode = SearchMode.EXACT;
}
```
