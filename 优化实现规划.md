```rust
const TECH_TERMS: &[&str] = &[
    // Rust 相关
    "内存安全",
    "零成本抽象",
    "所有权系统",
    "生命周期",
    "借用检查",
    "垃圾回收",
    
    // 编程通用
    "面向对象",
    "函数式编程",
    "泛型编程",
    "编译原理",
    "数据结构",
    "算法设计",
    
    // 中文搜索相关
    "中文分词",
    "自然语言处理",
    "搜索引擎",
];

// 修改 tokenize 函数
fn tokenize(text: &str) -> Vec<String> {
    let mut tokens = Vec::new();
    
    // 1. 先匹配专业术语
    for term in TECH_TERMS {
        if text.contains(term) {
            tokens.push(term.to_string());
        }
    }
    
    // 2. 再用 jieba 分词（排除已匹配的术语）
    let jieba_tokens = JIEBA.cut_for_search(text, false);
    tokens.extend(jieba_tokens.into_iter()
        .map(|s| s.to_string())
        .filter(|s| !TECH_TERMS.iter().any(|term| term.contains(s))));
    
    // 3. 添加 trigram
    add_trigrams(&mut tokens, text);
    
    tokens.sort();
    tokens.dedup();
    tokens
}
```

```rust
// crates/jieba-wasm/src/lib.rs
use jieba_rs::Jieba;
use wasm_bindgen::prelude::*;
use lazy_static::lazy_static;

lazy_static! {
    static ref JIEBA: Jieba = Jieba::new();
}

#[wasm_bindgen]
pub fn cut_for_search(text: &str) -> Vec<String> {
    JIEBA.cut_for_search(text, false)
        .into_iter()
        .map(|s| s.to_string())
        .collect()
}

#[wasm_bindgen]
pub fn cut(text: &str) -> Vec<String> {
    JIEBA.cut(text, false)
        .into_iter()
        .map(|s| s.to_string())
        .collect()
}

#[wasm_bindgen]
pub fn is_chinese(text: &str) -> bool {
    text.chars().any(|c| c as u32 >= 0x4E00 && c as u32 <= 0x9FFF)
}
```

```javascript
class SearchManager {
    constructor() {
        this.mode = SearchMode.FUZZY;  // 默认模糊
        this.modules = {
            jieba: null,
            fuzzy: null,
            regex: null,
        };
        this.loadingState = {
            jieba: 'loading',
            fuzzy: 'loading',
            regex: 'loading',
        };
    }
    
    async init() {
        // 1. 初始化基础搜索（立即）
        this.index = elasticlunr.Index.load(window.search.index);
        this.bindEvents();
        this.updateUI();
        
        // 2. 后台静默加载所有模块（不阻塞）
        this.backgroundLoad();
    }
    
    async backgroundLoad() {
        Promise.allSettled([
            this.loadJiebaWasm(),
            this.loadFuzzyModule(),
            this.loadRegexModule(),
        ]).then((results) => {
            const success = results.filter(r => r.status === 'fulfilled').length;
            console.log(`[Search] 后台加载完成：${success}/3 模块成功`);
            this.updateButtons();
            this.hideLoadingIndicator();
        });
    }
}
```

```javascript
// fuzzy_search.js
import Fuse from 'fuse.js';

let fuse = null;
let documents = null;

export async function initFuse(docs) {
    if (fuse) return fuse;
    
    fuse = new Fuse(docs, {
        keys: [
            { name: 'title', weight: 2 },
            { name: 'body', weight: 1 },
        ],
        threshold: 0.3,
        distance: 100,
        minMatchCharLength: 2,
        includeMatches: true,
    });
    
    documents = docs;
    return fuse;
}

export async function fuzzySearch(query) {
    const fuse = await initFuse(getDocuments());
    const results = fuse.search(query);
    
    return results.map(r => ({
        ref: r.item.id,
        score: r.score,
        doc: getDocument(r.item.id),
        matches: r.matches,
    }));
}
```

```javascript
setMode(newMode) {
    // 如果已经是当前模式，切换到精确
    if (this.mode === newMode) {
        this.mode = SearchMode.EXACT;
        this.updateButtons();
        this.search(this.searchInput.value);
        return;
    }
    
    // 检查模块是否已加载
    if (newMode === SearchMode.FUZZY && this.loadingState.fuzzy === 'loading') {
        this.showToast('✨ 模糊搜索加载中，请稍候...');
        return;
    }
    
    if (newMode === SearchMode.REGEX && this.loadingState.regex === 'loading') {
        this.showToast('.* 正则搜索加载中，请稍候...');
        return;
    }
    
    // 切换模式（自动互斥）
    this.mode = newMode;
    this.updateButtons();
    this.search(this.searchInput.value);
}
```
