### 1.2 添加拼音索引 🔧 (进行中)

**实现方案**:
```rust
// Cargo.toml
[dependencies]
pinyin = "0.10"

// search.rs
use pinyin::{Pinyin, to_pinyin_vec};

fn tokenize(text: &str) -> Vec<String> {
    let mut tokens = Vec::new();
    
    // 原有分词
    let jieba_tokens = JIEBA.cut_for_search(text, false);
    tokens.extend(jieba_tokens);
    
    // 添加拼音
    for word in &jieba_tokens {
        if is_chinese(word) {
            // 完整拼音：neicun
            let pinyin_full = to_pinyin_vec(word, false);
            tokens.push(pinyin_full.join(""));
            
            // 首字母：nc
            let pinyin_initials: String = pinyin_full
                .iter()
                .map(|p| p.initial().to_string())
                .collect();
            tokens.push(pinyin_initials);
        }
    }
    
    tokens
}
```

**验收标准**:
```bash
# 搜索拼音能找到中文内容
- 输入 "neicun" → 找到 "内存"
- 输入 "nj" → 找到 "内存"
- 输入 "yjgzj" → 找到 "一句话总结"
```

            }
        });
