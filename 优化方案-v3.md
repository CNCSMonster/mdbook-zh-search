#### 1.2 添加专业术语词典

**目标**：提高技术术语分词准确率

**实现**：
```rust
// 专业术语词典
const TECH_TERMS: &[&str] = &[
    // Rust 相关
    "内存安全",
    "零成本抽象",
    "所有权系统",
    "生命周期",
    "借用检查",
    
    // 编程通用
    "面向对象",
    "函数式编程",
    "泛型编程",
    "编译原理",
    
    // 中文搜索相关
    "中文分词",
    "自然语言处理",
    "搜索引擎",
];

fn tokenize(text: &str) -> Vec<String> {
    let mut tokens = Vec::new();
    
    // 1. 先匹配专业术语
    for term in TECH_TERMS {
        if text.contains(term) {
            tokens.push(term.to_string());
        }
    }
    
    // 2. 再用 jieba 分词
    let jieba_tokens = JIEBA.cut_for_search(text, false);
    tokens.extend(jieba_tokens.into_iter()
        .map(|s| s.to_string())
        .filter(|s| !TECH_TERMS.iter().any(|term| term.contains(s))));
    
    // 3. 添加 trigram
    add_trigrams(&mut tokens, text);
    
    tokens.sort();
    tokens.dedup();
    tokens
}
```

**验收标准**：
- "内存安全" 作为一个整体被索引
- "零成本抽象" 不被拆分为"零成本"+"抽象"

#### 1.3 运行时分析（jieba-rs WASM）

**目标**：搜索词也使用相同分词逻辑

**实现**：
```rust
// crates/jieba-wasm/src/lib.rs

use jieba_rs::Jieba;
use wasm_bindgen::prelude::*;
use lazy_static::lazy_static;

lazy_static! {
    static ref JIEBA: Jieba = Jieba::new();
}

#[wasm_bindgen]
pub fn cut_for_search(text: &str) -> Vec<String> {
    JIEBA.cut_for_search(text, false)
        .into_iter()
        .map(|s| s.to_string())
        .collect()
}
```

```javascript
// searcher.js
async function search(query) {
    // 中文搜索词需要分词
    if (isChinese(query)) {
        const jieba = await loadJiebaWasm();
        const tokens = jieba.cut_for_search(query);
        return index.search(tokens.join(' '), options);
    }
    
    // 英文直接搜索
    return index.search(query, options);
}
```

**验收标准**：
- 搜索"内存安全"能找到包含"内存"和"安全"的文档
- 搜索"一句话总结"能找到相关内容

### 阶段二：默认模糊搜索 + 后台加载（2-3 周）

#### 2.1 后台静默加载架构

**目标**：所有模块后台加载，用户无感知

**实现**：
```javascript
// searcher.js

class SearchManager {
    constructor() {
        // 默认模式：模糊搜索
        this.mode = SearchMode.FUZZY;
        
        this.modules = {
            jieba: null,
            fuzzy: null,
            regex: null,
        };
        
        this.loadingState = {
            jieba: 'loading',
            fuzzy: 'loading',
            regex: 'loading',
        };
    }
    
    async init() {
        // 1. 初始化基础搜索（立即）
        this.index = elasticlunr.Index.load(window.search.index);
        this.bindEvents();
        this.updateUI();
        
        // 2. 后台静默加载所有模块（不阻塞）
        this.backgroundLoad();
    }
    
    async backgroundLoad() {
        // 并行加载所有模块
        Promise.allSettled([
            this.loadJiebaWasm(),
            this.loadFuzzyModule(),
            this.loadRegexModule(),
        ]).then((results) => {
            const success = results.filter(r => r.status === 'fulfilled').length;
            console.log(`[Search] 后台加载完成：${success}/3 模块成功`);
            
            // 更新 UI，启用所有功能
            this.updateButtons();
            
            // 隐藏加载指示器
            this.hideLoadingIndicator();
        });
    }
    
    async loadJiebaWasm() {
        const module = await import('./jieba_wasm.js');
        this.modules.jieba = module;
        this.loadingState.jieba = 'ready';
    }
    
    async loadFuzzyModule() {
        const module = await import('./fuzzy_search.js');
        this.modules.fuzzy = module;
        this.loadingState.fuzzy = 'ready';
    }
    
    async loadRegexModule() {
        const wasm = await this.loadWasm('./wasm/regex_bg.wasm');
        this.modules.regex = wasm;
        this.loadingState.regex = 'ready';
    }
}
```

#### 2.2 默认模糊搜索

**目标**：开箱即用，支持拼写错误容错

**实现**：
```javascript
// 默认模式就是模糊搜索
const SearchMode = {
    FUZZY: 'fuzzy',     // 默认
    EXACT: 'exact',     // 精确
    REGEX: 'regex',     // 正则
};

class SearchManager {
    constructor() {
        this.mode = SearchMode.FUZZY;  // ✅ 默认模糊
    }
    
    async search(query) {
        if (!query) {
            this.hideResults();
            return;
        }
        
        let results;
        
        switch (this.mode) {
            case SearchMode.FUZZY:
                // 默认模糊搜索
                if (this.loadingState.fuzzy !== 'ready') {
                    // 模块还在加载中，降级为精确搜索
                    results = this.exactSearch(query);
                } else {
                    results = await this.modules.fuzzy.fuzzySearch(query);
                }
                break;
                
            case SearchMode.EXACT:
                results = this.exactSearch(query);
                break;
                
            case SearchMode.REGEX:
                results = this.modules.regex.search(query);
                break;
        }
        
        this.displayResults(results);
    }
}
```

**优势**：
- ✅ 支持拼写错误（"内存安权" → "内存安全"）
- ✅ 支持模糊匹配
- ✅ 用户体验更好
- ✅ 后台加载，无感知

#### 2.3 互斥的高级功能

**目标**：模糊和正则只能启用一个

**实现**：
```javascript
setMode(newMode) {
    // 如果已经是当前模式，切换到精确
    if (this.mode === newMode) {
        this.mode = SearchMode.EXACT;
        this.updateButtons();
        this.search(this.searchInput.value);
        return;
    }
    
    // 检查模块是否已加载
    if (newMode === SearchMode.FUZZY && this.loadingState.fuzzy === 'loading') {
        this.showToast('✨ 模糊搜索加载中，请稍候...');
        return;
    }
    
    if (newMode === SearchMode.REGEX && this.loadingState.regex === 'loading') {
        this.showToast('.* 正则搜索加载中，请稍候...');
        return;
    }
    
    // 切换模式（自动互斥）
    this.mode = newMode;
    this.updateButtons();
    this.search(this.searchInput.value);
}

updateButtons() {
    const buttons = {
        [SearchMode.FUZZY]: document.getElementById('btn-fuzzy'),
        [SearchMode.EXACT]: document.getElementById('btn-exact'),
        [SearchMode.REGEX]: document.getElementById('btn-regex'),
    };
    
    Object.entries(buttons).forEach(([mode, btn]) => {
        // 激活状态
        if (mode === this.mode) {
            btn.classList.add('active');
        } else {
            btn.classList.remove('active');
        }
        
        // 加载状态
        if (this.loadingState[mode] === 'loading') {
            btn.disabled = true;
            btn.textContent = this.getLoadingText(mode);
        } else {
            btn.disabled = false;
            btn.textContent = this.getDefaultText(mode);
        }
    });
}
```
