#### 1.2 添加专业术语词典

**目标**：提高技术术语分词准确率

**实现**：
```rust
// crates/mdbook-html/src/html_handlebars/search.rs

// 专业术语词典（按领域分类）
const TECH_TERMS: &[&str] = &[
    // Rust 相关
    "内存安全",
    "零成本抽象",
    "所有权系统",
    "生命周期",
    "借用检查",
    "垃圾回收",
    
    // 编程通用
    "面向对象",
    "函数式编程",
    "泛型编程",
    "编译原理",
    "数据结构",
    "算法设计",
    
    // 中文搜索相关
    "中文分词",
    "自然语言处理",
    "搜索引擎",
];

fn tokenize(text: &str) -> Vec<String> {
    let mut tokens = Vec::new();
    
    // 1. 先匹配专业术语
    for term in TECH_TERMS {
        if text.contains(term) {
            tokens.push(term.to_string());
        }
    }
    
    // 2. 再用 jieba 分词
    let jieba_tokens = JIEBA.cut_for_search(text, false);
    tokens.extend(jieba_tokens.into_iter()
        .map(|s| s.to_string())
        .filter(|s| !TECH_TERMS.iter().any(|term| term.contains(s))));
    
    // 3. 添加 trigram
    add_trigrams(&mut tokens, text);
    
    tokens.sort();
    tokens.dedup();
    tokens
}
```

**验收标准**：
- "内存安全" 作为一个整体被索引
- "零成本抽象" 不被拆分为"零成本"+"抽象"

#### 1.3 运行时分析（jieba-rs WASM）

**目标**：搜索词也使用相同分词逻辑

**架构**：
```
构建时：
  jieba-rs (Rust) → 分词 → 索引
  
运行时：
  jieba-rs WASM → 分词 → 搜索
```

**实现**：
```rust
// crates/jieba-wasm/src/lib.rs

use jieba_rs::Jieba;
use wasm_bindgen::prelude::*;
use lazy_static::lazy_static;

lazy_static! {
    static ref JIEBA: Jieba = Jieba::new();
}

#[wasm_bindgen]
pub fn cut_for_search(text: &str) -> Vec<String> {
    JIEBA.cut_for_search(text, false)
        .into_iter()
        .map(|s| s.to_string())
        .collect()
}

#[wasm_bindgen]
pub fn is_chinese(text: &str) -> bool {
    text.chars().any(|c| c as u32 >= 0x4E00 && c as u32 <= 0x9FFF)
}
```

```javascript
// searcher.js

let jiebaWasm = null;

async function loadJiebaWasm() {
    if (jiebaWasm) return jiebaWasm;
    
    // 按需加载
    const module = await import('./jieba_wasm.js');
    jiebaWasm = module;
    return module;
}

async function search(query) {
    // 中文搜索词需要分词
    if (isChinese(query)) {
        const jieba = await loadJiebaWasm();
        const tokens = jieba.cut_for_search(query);
        return index.search(tokens.join(' '), options);
    }
    
    // 英文直接搜索
    return index.search(query, options);
}
```

**验收标准**：
- 搜索"内存安全"能找到包含"内存"和"安全"的文档
- 搜索"一句话总结"能找到相关内容
- WASM 模块按需加载，不影响初始加载

### 阶段二：高级功能可选（2-3 周）

#### 2.1 按需加载架构

**目标**：高级功能不阻塞默认搜索

**实现**：
```javascript
// searcher.js

class SearchManager {
    constructor() {
        this.mode = 'exact';  // 'exact', 'fuzzy', 'regex'
        this.caseSensitive = false;
        this.modules = {};
        this.index = null;
    }
    
    async init() {
        // 初始化基础搜索（立即）
        this.index = elasticlunr.Index.load(window.search.index);
        this.bindEvents();
    }
    
    bindEvents() {
        // 搜索输入
        document.getElementById('search-input').oninput = (e) => {
            this.search(e.target.value);
        };
        
        // 模式切换
        document.getElementById('btn-fuzzy').onclick = () => {
            this.setMode('fuzzy');
        };
        
        document.getElementById('btn-regex').onclick = () => {
            this.setMode('regex');
        };
    }
    
    async setMode(mode) {
        this.mode = mode;
        
        // 按需加载模块
        if (mode === 'fuzzy' && !this.modules.fuzzy) {
            this.showLoading('加载模糊搜索...');
            const module = await import('./fuzzy_search.js');
            this.modules.fuzzy = module;
            this.hideLoading();
        }
        
        if (mode === 'regex' && !this.modules.regex) {
            this.showLoading('加载正则搜索...');
            const wasm = await this.loadWasm('./wasm/regex_bg.wasm');
            this.modules.regex = wasm;
            this.hideLoading();
        }
        
        this.search(document.getElementById('search-input').value);
    }
    
    async search(query) {
        if (!query) {
            this.hideResults();
            return;
        }
        
        let results;
        
        switch (this.mode) {
            case 'fuzzy':
                if (!this.modules.fuzzy) {
                    await this.setMode('fuzzy');
                    return;
                }
                results = this.modules.fuzzy.fuzzySearch(query);
                break;
                
            case 'regex':
                if (!this.modules.regex) {
                    await this.setMode('regex');
                    return;
                }
                results = this.modules.regex.search(query);
                break;
                
            default:
                // 精确搜索（已加载）
                results = this.index.search(query, options);
        }
        
        this.displayResults(results);
    }
    
    async loadWasm(path) {
        const response = await fetch(path);
        const { instance } = await WebAssembly.instantiateStreaming(response);
        return instance.exports;
    }
    
    showLoading(text) {
        // 显示加载状态
    }
    
    hideLoading() {
        // 隐藏加载状态
    }
    
    displayResults(results) {
        // 显示结果
    }
    
    hideResults() {
        // 隐藏结果
    }
}

// 启动
const searchManager = new SearchManager();
searchManager.init();
```

#### 2.2 模糊搜索（可选）

**目标**：支持拼写错误容错

```javascript
// fuzzy_search.js

import Fuse from 'fuse.js';

let fuse = null;

export async function initFuse(documents) {
    if (fuse) return fuse;
    
    fuse = new Fuse(documents, {
        keys: [
            { name: 'title', weight: 2 },
            { name: 'body', weight: 1 },
        ],
        threshold: 0.3,  // 0.0 = 精确，1.0 = 任意
        distance: 100,   // 匹配距离
        minMatchCharLength: 2,
        includeMatches: true,
    });
    
    return fuse;
}

export async function fuzzySearch(query) {
    const fuse = await initFuse(getDocuments());
    const results = fuse.search(query);
    
    return results.map(r => ({
        ref: r.item.id,
        score: r.score,
        doc: getDocument(r.item.id),
        matches: r.matches,
    }));
}
```

#### 2.3 正则搜索（可选）

**目标**：支持高级模式匹配

**实现**：
```javascript
// regex_search.js

export function regexSearch(pattern, documents, caseSensitive = false) {
    try {
        const flags = caseSensitive ? 'g' : 'gi';
        const regex = new RegExp(pattern, flags);
        
        const results = [];
        
        for (const [ref, doc] of Object.entries(documents)) {
            const allText = `${doc.title} ${doc.body}`;
            const matches = allText.match(regex);
            
            if (matches) {
                results.push({
                    ref,
                    score: matches.length,
                    doc,
                    matches: matches.slice(0, 5),
                });
            }
        }
        
        return results.sort((a, b) => b.score - a.score);
        
    } catch (e) {
        console.error('Invalid regex:', e);
        return [];
    }
}
```

**验收标准**：
- 搜索"/mem.*y/"能找到"memory"、"memroy"等
- 搜索"/[0-9]+/"能找到所有数字
- 无效正则表达式不崩溃，返回空结果

## 💡 关键决策

### 决策 1：默认搜索 vs 高级功能

**原则**：
- 默认搜索：快速、轻量、覆盖 90% 场景
- 高级功能：按需加载、不阻塞默认搜索

### 决策 2：WASM vs JavaScript

**原则**：
- 分词：WASM（保证一致性）
- 模糊搜索：JavaScript（Fuse.js 足够好）
- 正则搜索：原生 JavaScript（无需额外库）

### 决策 3：向后兼容

## 📝 book.toml 配置

```toml
[book]
title = "我的书籍"
authors = ["Author"]
language = "zh"

[output.html]
default-theme = "light"

[output.html.search]
enable = true

# 新增配置（可选）
[output.html.search.advanced]
# 启用高级搜索功能
enable-fuzzy = true
enable-regex = true
enable-runtime-tokenize = true

# 专业术语词典
custom-terms = ["内存安全", "零成本抽象"]
```
